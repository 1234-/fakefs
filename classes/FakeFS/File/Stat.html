      def initialize(file, __lstat = false)
        if !File.exists?(file)
          raise(Errno::ENOENT, "No such file or directory - #{file}")
        end

        @file      = file
        @fake_file = FileSystem.find(@file)
        @__lstat   = __lstat
        @ctime     = @fake_file.ctime
        @mtime     = @fake_file.mtime
      end
